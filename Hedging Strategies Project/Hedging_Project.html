#Functions

#Function to compute the price of an European call using the Black-Scholes-Merton formula
def BS(S, T, sigma, K, r):
    d1 = (np.log(S / K) + (r + 0.5 * sigma**2) * T) / (sigma * np.sqrt(T))
    d2 = d1 - sigma * np.sqrt(T)
    call_price = S * norm.cdf(d1) - K * np.exp(-r * T) * norm.cdf(d2)
    return call_price

#Function to compute the delta
def BS_delta(S, T, sigma, K, r):
    d1 = (np.log(S / K) + (r + 0.5 * sigma**2) * T) / (sigma * np.sqrt(T))
    return norm.cdf(d1)

#Function to compute the gamma
def BS_gamma(S, T, sigma, K, r):
    d1 = (np.log(S / K) + (r + 0.5 * sigma**2) * T) / (sigma * np.sqrt(T))
    return norm.pdf(d1) / (S * sigma * np.sqrt(T))

#Function to compute the vega
def BS_vega(S, T, sigma, K, r):
    d1 = (np.log(S / K) + (r + 0.5 * sigma**2) * T) / (sigma * np.sqrt(T))
    vega = S * np.sqrt(T) * norm.pdf(d1)
    return vega

#Newton-Raphson Function
def Newton2(f,xzero,prix,S,T,K,r,eps=0.0001,niter_max=15,h=0.01):
    x = xzero - f(xzero,prix,S,T,K,r)/((f(xzero+h,prix,S,T,K,r)-f(xzero,prix,S,T,K,r))/h)
    convergence = False
    nb_iteration = 0
    while abs(f(x,prix,S,T,K,r))>eps and nb_iteration <niter_max:
        nb_iteration +=1
        x = x - f(x,prix,S,T,K,r)/((f(x+h,prix,S,T,K,r)-f(x,prix,S,T,K,r))/h)

        if f(x,prix,S,T,K,r)<eps:
            convergence = True
    return x

#Dichotomie Function
def dichotomie(f,a,b,prix,S,T,K,r,eps=0.001,n=250):
    "f continuous f(a)<0<f(b)"
    iteration=0
    if f(a,prix,S,T,K,r)*f(b,prix,S,T,K,r)>0 :
        return ValueError("Il faut modifier le premier point de sorte que son image soit négative")
    if f(a,prix,S,T,K,r)>f(b,prix,S,T,K,r):
        a,b=b,a
    while b-a>eps and iteration<=n:
        iteration+=1
        c=(a+b)/2
        if f(c,prix,S,T,K,r)>0:
            b=c
        else:
            a=c
    return (a+b)/2

#Calibration of the implied volatility
#The goal of this function is to see the difference between the price in the variable 'prix' and the price using the BS function
def fct_ecart(vol,prix,S,T,K,r): # The variable 'prix' correspond to the price of an European call
    return BS(S,T,vol,K,r)-prix
